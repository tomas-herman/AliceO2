24 uint64_t calib_processing_helper::processRawData(o2::framework::InputRecord& inputs, std::unique_ptr<RawReaderCRU>& reader, bool useOldSubspec)
25 {
26  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{o2::header::gDataOriginTPC, "RAWDATA"}, Lifetime::Timeframe}};
27 
28  uint64_t activeSectors = 0;
29 
30  for (auto const& ref : InputRecordWalker(inputs, filter)) {
31  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
32 
33  // ---| extract hardware information to do the processing |---
34  const auto subSpecification = dh->subSpecification;
35  rdh_utils::FEEIDType feeID = (rdh_utils::FEEIDType)dh->subSpecification;
36  rdh_utils::FEEIDType cruID, linkID, endPoint;
37 
38  if (useOldSubspec) {
39  //---| old definition by Gvozden |---
40  cruID = (rdh_utils::FEEIDType)(subSpecification >> 16);
41  linkID = (rdh_utils::FEEIDType)((subSpecification + (subSpecification >> 8)) & 0xFF) - 1;
42  endPoint = (rdh_utils::FEEIDType)((subSpecification >> 8) & 0xFF) > 0;
43  } else {
44  //---| new definition by David |---
45  rdh_utils::getMapping(feeID, cruID, endPoint, linkID);
46  }
47 
48  uint64_t sector = cruID / 10;
49  activeSectors |= (0x1 << sector);
50 
51  const auto globalLinkID = linkID + endPoint * 12;
52 
53  // ---| update hardware information in the reader |---
54  reader->forceCRU(cruID);
55  reader->setLink(globalLinkID);
56 
57  LOGP(info, "Specifier: {}/{}/{}", dh->dataOrigin.as<std::string>(), dh->dataDescription.as<std::string>(), subSpecification);
58  LOGP(info, "Payload size: {}", dh->payloadSize);
59  LOGP(info, "CRU: {}; linkID: {}; endPoint: {}; globalLinkID: {}", cruID, linkID, endPoint, globalLinkID);
60 
61  // TODO: exception handling needed?
62  const gsl::span<const char> raw = inputs.get<gsl::span<char>>(ref);
63  o2::framework::RawParser parser(raw.data(), raw.size());
64 
65  // TODO: it would be better to have external event handling and then moving the event processing functionality to CalibRawBase and RawReader to not repeat it in other places
66  rawreader::ADCRawData rawData;
67  rawreader::GBTFrame gFrame;
68 
69  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
70 
71  const auto size = it.size();
72  auto data = it.data();
73  //LOGP(info, "Data size: {}", size);
74 
75  int iFrame = 0;
76  for (int i = 0; i < size; i += 16) {
77  gFrame.setFrameNumber(iFrame);
78  gFrame.setPacketNumber(iFrame / 508);
79  gFrame.readFromMemory(gsl::span<const o2::byte>(data + i, 16));
80 
81  // extract the half words from the 4 32-bit words
82  gFrame.getFrameHalfWords();
83 
84  gFrame.getAdcValues(rawData);
85  gFrame.updateSyncCheck(false);
86 
87  ++iFrame;
88  }
89  }
90 
91  reader->runADCDataCallback(rawData);
92  }
93 
94  return activeSectors;
95 }
