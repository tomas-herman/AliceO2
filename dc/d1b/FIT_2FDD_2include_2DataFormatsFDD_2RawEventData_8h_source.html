125  void fillTrigger(Triggers& trg)
126  {
127  trg.triggersignals = ((bool)orA << Triggers::bitA) |
128  ((bool)orC << Triggers::bitC) |
129  ((bool)vertex << Triggers::bitVertex) |
130  ((bool)cen << Triggers::bitCen) |

166  std::vector<char> to_vector(bool tcm)
167  {
168  constexpr int CRUWordSize = 16;
169 
170  std::vector<char> result(size() * CRUWordSize);
171  char* out = result.data();
172  if (!tcm) {
173  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
174  out += EventHeader::PayloadSize;
175  LOG(DEBUG) << " Write PM header: nWords: " << (int)mEventHeader.nGBTWords
176  << " orbit: " << int(mEventHeader.orbit)
177  << " BC: " << int(mEventHeader.bc)
178  << " size: " << result.size();
179  printHexEventHeader();
180  out += CRUWordSize - EventHeader::PayloadSize; // Padding enabled
181 
182  for (uint64_t i = 0; i < mEventHeader.nGBTWords; ++i) {
183  std::memcpy(out, &mEventData[2 * i], EventData::PayloadSizeFirstWord);
184  out += EventData::PayloadSizeFirstWord;
185  LOG(DEBUG) << " 1st word: Ch: " << std::setw(2) << mEventData[2 * i].channelID
186  << " charge: " << std::setw(4) << mEventData[2 * i].charge
187  << " time: " << std::setw(4) << mEventData[2 * i].time;
188  std::memcpy(out, &mEventData[2 * i + 1], EventData::PayloadSizeSecondWord);
189  out += EventData::PayloadSizeSecondWord;
190  LOG(DEBUG) << " 2nd word: Ch: " << std::setw(2) << mEventData[2 * i + 1].channelID
191  << " charge: " << std::setw(4) << mEventData[2 * i + 1].charge
192  << " time: " << std::setw(4) << mEventData[2 * i + 1].time;
193 
194  out += CRUWordSize - EventData::PayloadSizeSecondWord - EventData::PayloadSizeFirstWord;
195  printHexEventData(i);
196  }
197  } else {
198  // TCM data
199  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
200  out += EventHeader::PayloadSize;
201  LOG(DEBUG) << " Write TCM header: nWords: " << (int)mEventHeader.nGBTWords
202  << " orbit: " << int(mEventHeader.orbit)
203  << " BC: " << int(mEventHeader.bc)
204  << " size: " << result.size();
205  std::memcpy(out, &mTCMdata, sizeof(TCMdata));
206  out += sizeof(TCMdata);
207  // TODO: No TCM payload printing until the meaning of trigger bits and other flags is clarified
208  }
209  return result;
210  }

215  TCMdata mTCMdata;
