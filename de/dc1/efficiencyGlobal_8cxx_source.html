35  OutputObj<TH1F> pt{TH1F("pt", "pt", 20, 0., 10.)};

37  TH1F* efficiency = nullptr;

39  void init(InitContext&)
40  {
41  ccdb->setURL(url.value);
42  // Enabling object caching, otherwise each call goes to the CCDB server
43  ccdb->setCaching(true);
44  ccdb->setLocalObjectValidityChecking();
45  // Not later than now, will be replaced by the value of the train creation
46  // This avoids that users can replace objects **while** a train is running
47  ccdb->setCreatedNotAfter(nolaterthan.value);
48  LOGF(info, "Getting object %s", path.value.data());
49  efficiency = ccdb->getForTimeStamp<TH1F>(path.value, nolaterthan.value);
50  if (!efficiency) {
51  LOGF(fatal, "Efficiency object not found!");
52  }
53  }

55  void process(aod::Collision const& collision, aod::Tracks const& tracks)
56  {
57  for (auto& track : tracks) {

59  //LOGF(info, "Efficiency %f for pt %f", efficiency->GetBinContent(efficiency->FindBin(track.pt())), track.pt());
60  }
61  }
62 };
