144 template <typename S, typename T>
145 struct variant_helper {
146  static void set(S* store, T value) { *(reinterpret_cast<T*>(store)) = value; }
147  static void set(S* store, T values, size_t size)
148  {
149  *reinterpret_cast<T*>(store) = reinterpret_cast<T>(std::memcpy(std::malloc(size * sizeof(std::remove_pointer_t<T>)), reinterpret_cast<void*>(values), size * sizeof(std::remove_pointer_t<T>)));
150  }
151  static void reset(S* store, T values, size_t)
152  {
153  *reinterpret_cast<T*>(store) = values;
154  }
155 
156  static T get(const S* store) { return *(reinterpret_cast<const T*>(store)); }
157 };

188  template <typename T>

214  Variant(const Variant& other) : mType(other.mType)
215  {
216  // In case this is an array we need to duplicate it to avoid
217  // double deletion.
218  switch (mType) {
219  case variant_trait_v<const char*>:
220  mSize = other.mSize;
221  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
222  return;
223  case variant_trait_v<int*>:
224  mSize = other.mSize;
225  variant_helper<storage_t, int*>::set(&mStore, other.get<int*>(), mSize);
226  return;
227  case variant_trait_v<float*>:
228  mSize = other.mSize;
229  variant_helper<storage_t, float*>::set(&mStore, other.get<float*>(), mSize);
230  return;
231  case variant_trait_v<double*>:
232  mSize = other.mSize;
233  variant_helper<storage_t, double*>::set(&mStore, other.get<double*>(), mSize);
234  return;
235  case variant_trait_v<bool*>:
236  mSize = other.mSize;
237  variant_helper<storage_t, bool*>::set(&mStore, other.get<bool*>(), mSize);
238  return;
239  case variant_trait_v<std::string*>:

282  case variant_trait_v<int*>:
