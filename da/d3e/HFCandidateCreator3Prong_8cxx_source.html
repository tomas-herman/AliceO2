39  Produces<aod::HfCandProng3Base> rowCandidateBase;

41  Configurable<double> magneticField{"d_bz", 5., "magnetic field"};
42  Configurable<bool> b_propdca{"b_propdca", true, "create tracks version propagated to PCA"};
43  Configurable<double> d_maxr{"d_maxr", 200., "reject PCA's above this radius"};
44  Configurable<double> d_maxdzini{"d_maxdzini", 4., "reject (if>0) PCA candidate if tracks DZ exceeds threshold"};
45  Configurable<double> d_minparamchange{"d_minparamchange", 1.e-3, "stop iterations if largest change of any X is smaller than this"};
46  Configurable<double> d_minrelchi2change{"d_minrelchi2change", 0.9, "stop iterations is chi2/chi2old > this"};
47  Configurable<bool> b_dovalplots{"b_dovalplots", true, "do validation plots"};
48 
49  OutputObj<TH1F> hmass3{TH1F("hmass3", "3-prong candidates;inv. mass (#pi K #pi) (GeV/#it{c}^{2});entries", 500, 1.6, 2.1)};
50  OutputObj<TH1F> hCovPVXX{TH1F("hCovPVXX", "3-prong candidates;XX element of cov. matrix of prim. vtx position (cm^{2});entries", 100, 0., 1.e-4)};
51  OutputObj<TH1F> hCovSVXX{TH1F("hCovSVXX", "3-prong candidates;XX element of cov. matrix of sec. vtx position (cm^{2});entries", 100, 0., 0.2)};
52 
53  double massPi = RecoDecay::getMassPDG(kPiPlus);
54  double massK = RecoDecay::getMassPDG(kKPlus);
55  double massPiKPi{0.};

57  void process(aod::Collisions const& collisions,
58  aod::HfTrackIndexProng3 const& rowsTrackIndexProng3,
59  aod::BigTracks const& tracks)
60  {
61  // 3-prong vertex fitter
62  o2::vertexing::DCAFitterN<3> df;
63  df.setBz(magneticField);
64  df.setPropagateToPCA(b_propdca);
65  df.setMaxR(d_maxr);
66  df.setMaxDZIni(d_maxdzini);
67  df.setMinParamChange(d_minparamchange);
68  df.setMinRelChi2Change(d_minrelchi2change);
69  df.setUseAbsDCA(true);
70 
71  // loop over pairs of track indeces
72  for (const auto& rowTrackIndexProng3 : rowsTrackIndexProng3) {
73  auto trackParVar0 = getTrackParCov(rowTrackIndexProng3.index0());
74  auto trackParVar1 = getTrackParCov(rowTrackIndexProng3.index1());
75  auto trackParVar2 = getTrackParCov(rowTrackIndexProng3.index2());
76  auto collision = rowTrackIndexProng3.index0().collision();
77 
78  // reconstruct the 3-prong secondary vertex
79  if (df.process(trackParVar0, trackParVar1, trackParVar2) == 0) {
80  continue;
81  }
82  const auto& secondaryVertex = df.getPCACandidate();
83  auto chi2PCA = df.getChi2AtPCACandidate();
84  auto covMatrixPCA = df.calcPCACovMatrix().Array();
85  hCovSVXX->Fill(covMatrixPCA[0]); // FIXME: Calculation of errorDecayLength(XY) gives wrong values without this line.
86  trackParVar0 = df.getTrack(0);
87  trackParVar1 = df.getTrack(1);
88  trackParVar2 = df.getTrack(2);
89 
90  // get track momenta
91  array<float, 3> pvec0;
92  array<float, 3> pvec1;
93  array<float, 3> pvec2;
94  trackParVar0.getPxPyPzGlo(pvec0);
95  trackParVar1.getPxPyPzGlo(pvec1);
96  trackParVar2.getPxPyPzGlo(pvec2);
97 
98  // get track impact parameters
99  // This modifies track momenta!
100  auto primaryVertex = getPrimaryVertex(collision);
101  auto covMatrixPV = primaryVertex.getCov();
102  hCovPVXX->Fill(covMatrixPV[0]);
103  o2::dataformats::DCA impactParameter0;
104  o2::dataformats::DCA impactParameter1;
105  o2::dataformats::DCA impactParameter2;
106  trackParVar0.propagateToDCA(primaryVertex, magneticField, &impactParameter0);
107  trackParVar1.propagateToDCA(primaryVertex, magneticField, &impactParameter1);
108  trackParVar2.propagateToDCA(primaryVertex, magneticField, &impactParameter2);
109 
110  // get uncertainty of the decay length
111  double phi, theta;
112  getPointDirection(array{collision.posX(), collision.posY(), collision.posZ()}, secondaryVertex, phi, theta);
113  auto errorDecayLength = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, theta) + getRotatedCovMatrixXX(covMatrixPCA, phi, theta));
114  auto errorDecayLengthXY = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, 0.) + getRotatedCovMatrixXX(covMatrixPCA, phi, 0.));
115 
116  // fill candidate table rows
117  rowCandidateBase(collision.posX(), collision.posY(), collision.posZ(),
118  secondaryVertex[0], secondaryVertex[1], secondaryVertex[2],
119  errorDecayLength, errorDecayLengthXY,
120  chi2PCA,
121  pvec0[0], pvec0[1], pvec0[2],
122  pvec1[0], pvec1[1], pvec1[2],
123  pvec2[0], pvec2[1], pvec2[2],
124  impactParameter0.getY(), impactParameter1.getY(), impactParameter2.getY(),
125  std::sqrt(impactParameter0.getSigmaY2()), std::sqrt(impactParameter1.getSigmaY2()), std::sqrt(impactParameter2.getSigmaY2()),
126  rowTrackIndexProng3.index0Id(), rowTrackIndexProng3.index1Id(), rowTrackIndexProng3.index2Id(),
127  rowTrackIndexProng3.hfflag());
128 
129  // fill histograms
130  if (b_dovalplots) {
131  // calculate invariant mass
132  auto arrayMomenta = array{pvec0, pvec1, pvec2};
133  massPiKPi = RecoDecay::M(std::move(arrayMomenta), array{massPi, massK, massPi});
134  hmass3->Fill(massPiKPi);
135  }
136  }
137  }
138 };

147 struct HFCandidateCreator3ProngMC {
148  Produces<aod::HfCandProng3MCRec> rowMCMatchRec;
149  Produces<aod::HfCandProng3MCGen> rowMCMatchGen;
150 
151  void process(aod::HfCandProng3 const& candidates,
152  aod::BigTracksMC const& tracks,
153  aod::McParticles const& particlesMC)
154  {
155  int8_t sign = 0;
156  int8_t result = N3ProngDecays;
157 
158  // Match reconstructed candidates.
159  for (auto& candidate : candidates) {
160  //Printf("New rec. candidate");
161  result = N3ProngDecays;
162  auto arrayDaughters = array{candidate.index0_as<aod::BigTracksMC>(), candidate.index1_as<aod::BigTracksMC>(), candidate.index2_as<aod::BigTracksMC>()};
163 
164  // D± → π± K∓ π±
165  //Printf("Checking D± → π± K∓ π±");
166  if (RecoDecay::getMatchedMCRec(particlesMC, arrayDaughters, 411, array{+kPiPlus, -kKPlus, +kPiPlus}, true, &sign) > -1) {
167  result = sign * DPlusToPiKPi;
168  }
169 
170  // Λc± → p± K∓ π±
171  if (result == N3ProngDecays) {
172  //Printf("Checking Λc± → p± K∓ π±");
173  if (RecoDecay::getMatchedMCRec(particlesMC, std::move(arrayDaughters), 4122, array{+kProton, -kKPlus, +kPiPlus}, true, &sign) > -1) {
174  result = sign * LcToPKPi;
175  }
176  }
177 
178  rowMCMatchRec(result);
179  }
180 
181  // Match generated particles.
182  for (auto& particle : particlesMC) {
183  //Printf("New gen. candidate");
184  result = N3ProngDecays;
185 
186  // D± → π± K∓ π±
187  //Printf("Checking D± → π± K∓ π±");
188  if (RecoDecay::isMatchedMCGen(particlesMC, particle, 411, array{+kPiPlus, -kKPlus, +kPiPlus}, true, &sign)) {
189  result = sign * DPlusToPiKPi;
190  }
191 
192  // Λc± → p± K∓ π±
193  if (result == N3ProngDecays) {
194  //Printf("Checking Λc± → p± K∓ π±");
195  if (RecoDecay::isMatchedMCGen(particlesMC, particle, 4122, array{+kProton, -kKPlus, +kPiPlus}, true, &sign)) {
196  result = sign * LcToPKPi;
197  }
198  }
199 
200  rowMCMatchGen(result);
201  }
202  }
203 };
