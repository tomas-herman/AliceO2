55 RawErrorType_t RawDecoder::readChannels()
56 {
57  mChannelsInitialized = false;
58  auto& header = mRawReader.getRawHeader();
59  short ddl = o2::raw::RDHUtils::getFEEID(header); //Current fee/ddl
60 
61  auto& payloadwords = mRawReader.getPayload();
62  //start reading from the end
63  auto currentWord = payloadwords.rbegin();
64  while (currentWord != payloadwords.rend()) {
65  SegMarkerWord sw = {*currentWord++}; //first get value, then increment
66  if (sw.marker != 2736) { //error
67  mErrors.emplace_back(ddl, 17, 2, 0, kSEGMENT_HEADER_ERROR); //add error for non-existing row
68  //try adding this as padWord
69  addDigit(sw.mDataWord, ddl);
70  continue;
71  }
72  short nSegWords = sw.nwords;
73  short currentRow = sw.row;
74  short nEoE = 0;
75  while (nSegWords > 0 && currentWord != payloadwords.rend()) {
76  EoEWord ew = {*currentWord++};
77  nSegWords--;
78  if (ew.checkbit != 1) { //error
79  LOG(ERROR) << " error EoE, ddl" << ddl << " row " << currentRow;
80  mErrors.emplace_back(ddl, currentRow, 11, 0, kEOE_HEADER_ERROR); //add error
81  //try adding this as padWord
82  addDigit(ew.mDataWord, ddl);
83  continue;
84  }
85  nEoE++;
86  short nEoEwords = ew.nword;
87  short currentDilogic = ew.dilogic;
88  if (ew.row != currentRow) {
89  LOG(ERROR) << "Row in EoE=" << ew.row << " != expected row " << currentRow;
90  mErrors.emplace_back(ddl, currentRow, currentDilogic, 0, kEOE_HEADER_ERROR); //add error
91  //try adding this as padWord
92  addDigit(ew.mDataWord, ddl);
93  continue;
94  }
95  if (currentDilogic < 0 || currentDilogic > 10) {
96  LOG(ERROR) << "wrong Dilogic in EoE=" << currentDilogic;
97  mErrors.emplace_back(ddl, currentRow, currentDilogic, 0, kEOE_HEADER_ERROR); //add error
98  //try adding this as padWord
99  addDigit(ew.mDataWord, ddl);
100  continue;
101  }
102  while (nEoEwords > 0 && currentWord != payloadwords.rend()) {
103  PadWord pad = {*currentWord++};
104  nEoEwords--;
105  nSegWords--;
106  if (pad.zero != 0) {
107  LOG(ERROR) << "bad pad, word=" << pad.mDataWord;
108  mErrors.emplace_back(ddl, currentRow, currentDilogic, 49, kPADERROR); //add error and skip word
109  continue;
110  }
111  //check paw/pad indexes
112  if (pad.row != currentRow || pad.dilogic != currentDilogic) {
113  LOG(ERROR) << "RawPad " << pad.row << " != currentRow=" << currentRow << "dilogicPad=" << pad.dilogic << "!= currentDilogic=" << currentDilogic;
114  mErrors.emplace_back(ddl, short(pad.row), short(pad.dilogic), short(pad.address), kPadAddress); //add error and skip word
115  //do not skip, try adding using info from pad
116  }
117  addDigit(pad.mDataWord, ddl);
118  } //pads in EoE
119  if (nEoE % 10 == 0) { // kNDilogic = 10; ///< Number of dilogic per row
120  if (currentWord != payloadwords.rend()) { //Read row HEader
121  RowMarkerWord rw = {*currentWord++};
122  nSegWords--;
123  currentRow--;
124  if (rw.marker != 13992) {
125  LOG(ERROR) << "Error in row marker:" << rw.marker << "row header word=" << rw.mDataWord;
126  mErrors.emplace_back(ddl, currentRow, 11, 0, kPadAddress); //add error and skip word
127  //try adding digit assuming this is pad word
128  addDigit(rw.mDataWord, ddl);
129  }
130  }
131  }
132  } // in Segment
133  }
134  mChannelsInitialized = true;
135  return kOK;
136 }

154 void RawDecoder::addDigit(uint32_t w, short ddl)
155 {
156 
157  PadWord pad = {w};
158  if (pad.zero != 0) {
159  return;
160  }
161  short rowPad = pad.row;
162  short dilogicPad = pad.dilogic;
163  short hw = pad.address;
164  unsigned short absId;
165  o2::cpv::Geometry::hwaddressToAbsId(ddl, rowPad, dilogicPad, hw, absId);
166 
167  AddressCharge ac = {0};
168  ac.Address = absId;
169  ac.Charge = pad.charge;
170  mDigits.push_back(ac.mDataWord);
171 }
