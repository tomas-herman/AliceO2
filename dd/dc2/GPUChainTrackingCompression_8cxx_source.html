27 int GPUChainTracking::RunTPCCompression()
28 {
29 #ifdef HAVE_O2HEADERS
30  mRec->PushNonPersistentMemory();
31  RecoStep myStep = RecoStep::TPCCompression;
32  bool doGPU = GetRecoStepsGPU() & RecoStep::TPCCompression;
33  GPUTPCCompression& Compressor = processors()->tpcCompressor;
34  GPUTPCCompression& CompressorShadow = doGPU ? processorsShadow()->tpcCompressor : Compressor;
35  const auto& threadContext = GetThreadContext();
36  if (mPipelineFinalizationCtx && GetProcessingSettings().doublePipelineClusterizer) {
37  RecordMarker(&mEvents->single, 0);
38  }
39  Compressor.mNMaxClusterSliceRow = 0;
40  for (unsigned int i = 0; i < NSLICES; i++) {
41  for (unsigned int j = 0; j < GPUCA_ROW_COUNT; j++) {
42  if (mIOPtrs.clustersNative->nClusters[i][j] > Compressor.mNMaxClusterSliceRow) {
43  Compressor.mNMaxClusterSliceRow = mIOPtrs.clustersNative->nClusters[i][j];
44  }
45  }
46  }
47 
48  if (ProcessingSettings().tpcCompressionGatherMode == 3) {
49  mRec->AllocateVolatileDeviceMemory(0); // make future device memory allocation volatile
50  }
51  SetupGPUProcessor(&Compressor, true);
52  new (Compressor.mMemory) GPUTPCCompression::memory;
53 
54  WriteToConstantMemory(myStep, (char*)&processors()->tpcCompressor - (char*)processors(), &CompressorShadow, sizeof(CompressorShadow), 0);
55  TransferMemoryResourcesToGPU(myStep, &Compressor, 0);
56  runKernel<GPUMemClean16>(GetGridAutoStep(0, RecoStep::TPCCompression), krnlRunRangeNone, krnlEventNone, CompressorShadow.mClusterStatus, Compressor.mMaxClusters * sizeof(CompressorShadow.mClusterStatus[0]));
57  runKernel<GPUTPCCompressionKernels, GPUTPCCompressionKernels::step0attached>(GetGridAuto(0), krnlRunRangeNone, krnlEventNone);
58  runKernel<GPUTPCCompressionKernels, GPUTPCCompressionKernels::step1unattached>(GetGridAuto(0), krnlRunRangeNone, krnlEventNone);
59  TransferMemoryResourcesToHost(myStep, &Compressor, 0);
60 #ifdef GPUCA_TPC_GEOMETRY_O2
61  if (mPipelineFinalizationCtx && GetProcessingSettings().doublePipelineClusterizer) {
62  SynchronizeEvents(&mEvents->single);
63  ReleaseEvent(&mEvents->single);
64  ((GPUChainTracking*)GetNextChainInQueue())->RunTPCClusterizer_prepare(false);
65  ((GPUChainTracking*)GetNextChainInQueue())->mCFContext->ptrClusterNativeSave = processorsShadow()->ioPtrs.clustersNative;
66  }
67 #endif
68  SynchronizeStream(0);
69  o2::tpc::CompressedClusters* O = Compressor.mOutput;
70  memset((void*)O, 0, sizeof(*O));
71  O->nTracks = Compressor.mMemory->nStoredTracks;
72  O->nAttachedClusters = Compressor.mMemory->nStoredAttachedClusters;
73  O->nUnattachedClusters = Compressor.mMemory->nStoredUnattachedClusters;
74  O->nAttachedClustersReduced = O->nAttachedClusters - O->nTracks;
75  O->nSliceRows = NSLICES * GPUCA_ROW_COUNT;
76  O->nComppressionModes = param().rec.tpcCompressionModes;
77  size_t outputSize = AllocateRegisteredMemory(Compressor.mMemoryResOutputHost, mSubOutputControls[GPUTrackingOutputs::getIndex(&GPUTrackingOutputs::compressedClusters)]);
78  Compressor.mOutputFlat->set(outputSize, *Compressor.mOutput);
79  char* hostFlatPtr = (char*)Compressor.mOutput->qTotU; // First array as allocated in GPUTPCCompression::SetPointersCompressedClusters
80  size_t copySize = 0;
81  if (ProcessingSettings().tpcCompressionGatherMode == 3) {
82  CompressorShadow.mOutputA = Compressor.mOutput;
83  copySize = AllocateRegisteredMemory(Compressor.mMemoryResOutputGPU); // We overwrite Compressor.mOutput with the allocated output pointers on the GPU
84  }
85  const o2::tpc::CompressedClustersPtrs* P = nullptr;
86  HighResTimer* gatherTimer = nullptr;
87  int outputStream = 0;
88  if (ProcessingSettings().doublePipeline) {
89  SynchronizeStream(mRec->NStreams() - 2); // Synchronize output copies running in parallel from memory that might be released, only the following async copy from stacked memory is safe after the chain finishes.
90  outputStream = mRec->NStreams() - 2;
91  }
92 
93  if (ProcessingSettings().tpcCompressionGatherMode >= 2) {
94  if (ProcessingSettings().tpcCompressionGatherMode == 2) {
95  void* devicePtr = mRec->getGPUPointer(Compressor.mOutputFlat);
96  if (devicePtr != Compressor.mOutputFlat) {
97  CompressedClustersPtrs& ptrs = *Compressor.mOutput; // We need to update the ptrs with the gpu-mapped version of the host address space
98  for (unsigned int i = 0; i < sizeof(ptrs) / sizeof(void*); i++) {
99  reinterpret_cast<char**>(&ptrs)[i] = reinterpret_cast<char**>(&ptrs)[i] + (reinterpret_cast<char*>(devicePtr) - reinterpret_cast<char*>(Compressor.mOutputFlat));
100  }
101  }
102  }
103  TransferMemoryResourcesToGPU(myStep, &Compressor, outputStream);
104  constexpr unsigned int nBlocksDefault = 2;
105  constexpr unsigned int nBlocksMulti = 1 + 2 * 200;
106  switch (ProcessingSettings().tpcCompressionGatherModeKernel) {
107  case 0:
108  runKernel<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::unbuffered>(GetGridBlkStep(nBlocksDefault, outputStream, RecoStep::TPCCompression), krnlRunRangeNone, krnlEventNone);
109  getKernelTimer<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::unbuffered>(RecoStep::TPCCompression, 0, outputSize);
110  break;
111  case 1:
112  runKernel<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::buffered32>(GetGridBlkStep(nBlocksDefault, outputStream, RecoStep::TPCCompression), krnlRunRangeNone, krnlEventNone);
113  getKernelTimer<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::buffered32>(RecoStep::TPCCompression, 0, outputSize);
114  break;
115  case 2:
116  runKernel<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::buffered64>(GetGridBlkStep(nBlocksDefault, outputStream, RecoStep::TPCCompression), krnlRunRangeNone, krnlEventNone);
117  getKernelTimer<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::buffered64>(RecoStep::TPCCompression, 0, outputSize);
118  break;
119  case 3:
120  runKernel<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::buffered128>(GetGridBlkStep(nBlocksDefault, outputStream, RecoStep::TPCCompression), krnlRunRangeNone, krnlEventNone);
121  getKernelTimer<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::buffered128>(RecoStep::TPCCompression, 0, outputSize);
122  break;
123  case 4:
124 
125  static_assert((nBlocksMulti & 1) && nBlocksMulti >= 3);
126  runKernel<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::multiBlock>(GetGridBlkStep(nBlocksMulti, outputStream, RecoStep::TPCCompression), krnlRunRangeNone, krnlEventNone);
127  getKernelTimer<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::multiBlock>(RecoStep::TPCCompression, 0, outputSize);
128  break;
129  default:
130  GPUError("Invalid compression kernel selected.");
131  return 1;
132  }
133  if (ProcessingSettings().tpcCompressionGatherMode == 3) {
134  RecordMarker(&mEvents->stream[outputStream], outputStream);
135  char* deviceFlatPts = (char*)Compressor.mOutput->qTotU;
136  if (GetProcessingSettings().doublePipeline) {
137  const size_t blockSize = CAMath::nextMultipleOf<1024>(copySize / 30);
138  const unsigned int n = (copySize + blockSize - 1) / blockSize;
139  for (unsigned int i = 0; i < n; i++) {
140  GPUMemCpy(myStep, hostFlatPtr + i * blockSize, deviceFlatPts + i * blockSize, CAMath::Min(blockSize, copySize - i * blockSize), outputStream, false);
141  }
142  } else {
143  GPUMemCpy(myStep, hostFlatPtr, deviceFlatPts, copySize, outputStream, false);
144  }
145  }
146  } else {
147  char direction = 0;
148  if (ProcessingSettings().tpcCompressionGatherMode == 0) {
149  P = &CompressorShadow.mPtrs;
150  } else if (ProcessingSettings().tpcCompressionGatherMode == 1) {
151  P = &Compressor.mPtrs;
152  direction = -1;
153  gatherTimer = &getTimer<GPUTPCCompressionKernels>("GPUTPCCompression_GatherOnCPU", 0);
154  gatherTimer->Start();
155  }
156  GPUMemCpyAlways(myStep, O->nSliceRowClusters, P->nSliceRowClusters, NSLICES * GPUCA_ROW_COUNT * sizeof(O->nSliceRowClusters[0]), outputStream, direction);
157  GPUMemCpyAlways(myStep, O->nTrackClusters, P->nTrackClusters, O->nTracks * sizeof(O->nTrackClusters[0]), outputStream, direction);
158  SynchronizeStream(outputStream);
159  unsigned int offset = 0;
160  for (unsigned int i = 0; i < NSLICES; i++) {
161  for (unsigned int j = 0; j < GPUCA_ROW_COUNT; j++) {
162  GPUMemCpyAlways(myStep, O->qTotU + offset, P->qTotU + mIOPtrs.clustersNative->clusterOffset[i][j], O->nSliceRowClusters[i * GPUCA_ROW_COUNT + j] * sizeof(O->qTotU[0]), outputStream, direction);
163  GPUMemCpyAlways(myStep, O->qMaxU + offset, P->qMaxU + mIOPtrs.clustersNative->clusterOffset[i][j], O->nSliceRowClusters[i * GPUCA_ROW_COUNT + j] * sizeof(O->qMaxU[0]), outputStream, direction);
164  GPUMemCpyAlways(myStep, O->flagsU + offset, P->flagsU + mIOPtrs.clustersNative->clusterOffset[i][j], O->nSliceRowClusters[i * GPUCA_ROW_COUNT + j] * sizeof(O->flagsU[0]), outputStream, direction);
165  GPUMemCpyAlways(myStep, O->padDiffU + offset, P->padDiffU + mIOPtrs.clustersNative->clusterOffset[i][j], O->nSliceRowClusters[i * GPUCA_ROW_COUNT + j] * sizeof(O->padDiffU[0]), outputStream, direction);
166  GPUMemCpyAlways(myStep, O->timeDiffU + offset, P->timeDiffU + mIOPtrs.clustersNative->clusterOffset[i][j], O->nSliceRowClusters[i * GPUCA_ROW_COUNT + j] * sizeof(O->timeDiffU[0]), outputStream, direction);
167  GPUMemCpyAlways(myStep, O->sigmaPadU + offset, P->sigmaPadU + mIOPtrs.clustersNative->clusterOffset[i][j], O->nSliceRowClusters[i * GPUCA_ROW_COUNT + j] * sizeof(O->sigmaPadU[0]), outputStream, direction);
168  GPUMemCpyAlways(myStep, O->sigmaTimeU + offset, P->sigmaTimeU + mIOPtrs.clustersNative->clusterOffset[i][j], O->nSliceRowClusters[i * GPUCA_ROW_COUNT + j] * sizeof(O->sigmaTimeU[0]), outputStream, direction);
169  offset += O->nSliceRowClusters[i * GPUCA_ROW_COUNT + j];
170  }
171  }
172  offset = 0;
173  for (unsigned int i = 0; i < O->nTracks; i++) {
174  GPUMemCpyAlways(myStep, O->qTotA + offset, P->qTotA + Compressor.mAttachedClusterFirstIndex[i], O->nTrackClusters[i] * sizeof(O->qTotA[0]), outputStream, direction);
175  GPUMemCpyAlways(myStep, O->qMaxA + offset, P->qMaxA + Compressor.mAttachedClusterFirstIndex[i], O->nTrackClusters[i] * sizeof(O->qMaxA[0]), outputStream, direction);
176  GPUMemCpyAlways(myStep, O->flagsA + offset, P->flagsA + Compressor.mAttachedClusterFirstIndex[i], O->nTrackClusters[i] * sizeof(O->flagsA[0]), outputStream, direction);
177  GPUMemCpyAlways(myStep, O->sigmaPadA + offset, P->sigmaPadA + Compressor.mAttachedClusterFirstIndex[i], O->nTrackClusters[i] * sizeof(O->sigmaPadA[0]), outputStream, direction);
178  GPUMemCpyAlways(myStep, O->sigmaTimeA + offset, P->sigmaTimeA + Compressor.mAttachedClusterFirstIndex[i], O->nTrackClusters[i] * sizeof(O->sigmaTimeA[0]), outputStream, direction);
179 
180  // First index stored with track
181  GPUMemCpyAlways(myStep, O->rowDiffA + offset - i, P->rowDiffA + Compressor.mAttachedClusterFirstIndex[i] + 1, (O->nTrackClusters[i] - 1) * sizeof(O->rowDiffA[0]), outputStream, direction);
182  GPUMemCpyAlways(myStep, O->sliceLegDiffA + offset - i, P->sliceLegDiffA + Compressor.mAttachedClusterFirstIndex[i] + 1, (O->nTrackClusters[i] - 1) * sizeof(O->sliceLegDiffA[0]), outputStream, direction);
183  GPUMemCpyAlways(myStep, O->padResA + offset - i, P->padResA + Compressor.mAttachedClusterFirstIndex[i] + 1, (O->nTrackClusters[i] - 1) * sizeof(O->padResA[0]), outputStream, direction);
184  GPUMemCpyAlways(myStep, O->timeResA + offset - i, P->timeResA + Compressor.mAttachedClusterFirstIndex[i] + 1, (O->nTrackClusters[i] - 1) * sizeof(O->timeResA[0]), outputStream, direction);
185  offset += O->nTrackClusters[i];
186  }
187  GPUMemCpyAlways(myStep, O->qPtA, P->qPtA, O->nTracks * sizeof(O->qPtA[0]), outputStream, direction);
188  GPUMemCpyAlways(myStep, O->rowA, P->rowA, O->nTracks * sizeof(O->rowA[0]), outputStream, direction);
189  GPUMemCpyAlways(myStep, O->sliceA, P->sliceA, O->nTracks * sizeof(O->sliceA[0]), outputStream, direction);
190  GPUMemCpyAlways(myStep, O->timeA, P->timeA, O->nTracks * sizeof(O->timeA[0]), outputStream, direction);
191  GPUMemCpyAlways(myStep, O->padA, P->padA, O->nTracks * sizeof(O->padA[0]), outputStream, direction);
192  }
193  if (ProcessingSettings().tpcCompressionGatherMode == 1) {
194  gatherTimer->Stop();
195  }
196  mIOPtrs.tpcCompressedClusters = Compressor.mOutputFlat;
197  if (ProcessingSettings().tpcCompressionGatherMode == 3) {
198  SynchronizeEvents(&mEvents->stream[outputStream]);
199  ReleaseEvent(&mEvents->stream[outputStream]);
200  mRec->ReturnVolatileDeviceMemory();
201  }
202 
203  if (mPipelineFinalizationCtx == nullptr) {
204  SynchronizeStream(outputStream);
205  } else {
206  ((GPUChainTracking*)GetNextChainInQueue())->mRec->BlockStackedMemory(mRec);
207  }
208  mRec->PopNonPersistentMemory(RecoStep::TPCCompression);
209 #endif
210  return 0;
211 }

213 int GPUChainTracking::RunTPCDecompression()
214 {
215 #ifdef HAVE_O2HEADERS
216  const auto& threadContext = GetThreadContext();
217  TPCClusterDecompressor decomp;
218  auto allocator = [this](size_t size) {
219  this->mInputsHost->mNClusterNative = this->mInputsShadow->mNClusterNative = size;
220  this->AllocateRegisteredMemory(this->mInputsHost->mResourceClusterNativeOutput, this->mSubOutputControls[GPUTrackingOutputs::getIndex(&GPUTrackingOutputs::clustersNative)]);
221  return this->mInputsHost->mPclusterNativeOutput;
222  };
223  auto& gatherTimer = getTimer<TPCClusterDecompressor>("TPCDecompression", 0);
224  gatherTimer.Start();
225  if (decomp.decompress(mIOPtrs.tpcCompressedClusters, *mClusterNativeAccess, allocator, param())) {
226  GPUError("Error decompressing clusters");
227  return 1;
228  }
229  gatherTimer.Stop();
230  mIOPtrs.clustersNative = mClusterNativeAccess.get();
231  if (mRec->IsGPU()) {
232  AllocateRegisteredMemory(mInputsHost->mResourceClusterNativeBuffer);
233  processorsShadow()->ioPtrs.clustersNative = mInputsShadow->mPclusterNativeAccess;
234  WriteToConstantMemory(RecoStep::TPCDecompression, (char*)&processors()->ioPtrs - (char*)processors(), &processorsShadow()->ioPtrs, sizeof(processorsShadow()->ioPtrs), 0);
235  *mInputsHost->mPclusterNativeAccess = *mIOPtrs.clustersNative;
236  mInputsHost->mPclusterNativeAccess->clustersLinear = mInputsShadow->mPclusterNativeBuffer;
237  mInputsHost->mPclusterNativeAccess->setOffsetPtrs();
238  GPUMemCpy(RecoStep::TPCDecompression, mInputsShadow->mPclusterNativeBuffer, mIOPtrs.clustersNative->clustersLinear, sizeof(mIOPtrs.clustersNative->clustersLinear[0]) * mIOPtrs.clustersNative->nClustersTotal, 0, true);
239  TransferMemoryResourceLinkToGPU(RecoStep::TPCDecompression, mInputsHost->mResourceClusterNativeAccess, 0);
240  SynchronizeStream(0);
241  }
242 #endif
243  return 0;
244 }
