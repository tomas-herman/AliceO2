116 inline const Float_t Tracker::getDistanceToSeed(const Cluster& cluster1, const Cluster& cluster2, const Cluster& cluster) const
117 {
118  // the seed is between "cluster1" and "cluster2" and cuts the plane
119  // of the "cluster" at a distance dR from it
120  Float_t dxSeed, dySeed, dzSeed, invdzSeed, dz, dR2, xSeed, ySeed;
121  dxSeed = cluster2.getX() - cluster1.getX();
122  dySeed = cluster2.getY() - cluster1.getY();
123  dzSeed = cluster2.getZ() - cluster1.getZ();
124  dz = cluster.getZ() - cluster1.getZ();
125  invdzSeed = dz / dzSeed;
126  xSeed = cluster1.getX() + dxSeed * invdzSeed;
127  ySeed = cluster1.getY() + dySeed * invdzSeed;
128  dR2 = (cluster.getX() - xSeed) * (cluster.getX() - xSeed) + (cluster.getY() - ySeed) * (cluster.getY() - ySeed);
129  return dR2;
130 }

141 inline const Float_t Tracker::getCellDeviation(const Cell& cell1, const Cell& cell2) const
142 {
143  Float_t cell1dx = cell1.getX2() - cell1.getX1();
144  Float_t cell1dy = cell1.getY2() - cell1.getY1();
145  Float_t cell1dz = cell1.getZ2() - cell1.getZ1();
146 
147  Float_t cell2dx = cell2.getX2() - cell2.getX1();
148  Float_t cell2dy = cell2.getY2() - cell2.getY1();
149  Float_t cell2dz = cell2.getZ2() - cell2.getZ1();
150 
151  Float_t cell1mod = std::sqrt(cell1dx * cell1dx + cell1dy * cell1dy + cell1dz * cell1dz);
152  Float_t cell2mod = std::sqrt(cell2dx * cell2dx + cell2dy * cell2dy + cell2dz * cell2dz);
153 
154  Float_t cosAngle = (cell1dx * cell2dx + cell1dy * cell2dy + cell1dz * cell2dz) / (cell1mod * cell2mod);
155 
156  return std::acos(cosAngle);
157 }

160 inline const Bool_t Tracker::getCellsConnect(const Cell& cell1, const Cell& cell2) const
161 {
162  Float_t cell1x2 = cell1.getX2();
163  Float_t cell1y2 = cell1.getY2();
164  Float_t cell2x1 = cell2.getX1();
165  Float_t cell2y1 = cell2.getY1();
166  Float_t dx = cell1x2 - cell2x1;
167  Float_t dy = cell1y2 - cell2y1;
168  Float_t dr2 = dx * dx + dy * dy;
169 
170  if (dr2 > (constants::mft::Resolution * constants::mft::Resolution)) {
171  return kFALSE;
172  }
173  return kTRUE;
174 }

177 template <class T>
178 inline void Tracker::computeTracksMClabels(const T& tracks)
179 {
181  for (auto& track : tracks) {
182  MCCompLabel maxOccurrencesValue{-1, -1, -1, false};
183  int count{0};
184  bool isFakeTrack{false};
185  auto nClusters = track.getNumberOfPoints();
186  for (int iCluster = 0; iCluster < nClusters; ++iCluster) {
187  const MCCompLabel& currentLabel = track.getMCCompLabels()[iCluster];
188  if (currentLabel == maxOccurrencesValue) {
189  ++count;
190  } else {
191  if (count != 0) { // only in the first iteration count can be 0 at this point
192  --count;
193  }
194  if (count == 0) {
195  maxOccurrencesValue = currentLabel;
196  count = 1;
197  }
198  }
199  }
200  count = 0;
201  for (int iCluster = 0; iCluster < nClusters; ++iCluster) {
202  if (track.getMCCompLabels()[iCluster] == maxOccurrencesValue) {
203  count++;
204  }
205  }
206 
207  auto labelratio = 1.0 * count / nClusters;
208  if (labelratio >= 0.8) {
209  } else {
210  isFakeTrack = true;
211  maxOccurrencesValue.setFakeFlag();
212  }
213  mTrackLabels.addElement(mTrackLabels.getIndexedSize(), maxOccurrencesValue);
214  }
215 }
