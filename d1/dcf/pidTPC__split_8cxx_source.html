39 template <o2::track::PID::ID pid_type, typename table>
40 struct pidTPCTaskPerParticle {
41  using Trks = soa::Join<aod::Tracks, aod::TracksExtra>;
42  using Coll = aod::Collisions;
43  Produces<table> tpcpid;
44  DetectorResponse resp;
45  Service<o2::ccdb::BasicCCDBManager> ccdb;
46  Configurable<std::string> paramfile{"param-file", "", "Path to the parametrization object, if emtpy the parametrization is not taken from file"};
47  Configurable<std::string> signalname{"param-signal", "BetheBloch", "Name of the parametrization for the expected signal, used in both file and CCDB mode"};
48  Configurable<std::string> sigmaname{"param-sigma", "TPCReso", "Name of the parametrization for the expected sigma, used in both file and CCDB mode"};
49  Configurable<std::string> url{"ccdb-url", "http://ccdb-test.cern.ch:8080", "url of the ccdb repository"};
50  Configurable<long> timestamp{"ccdb-timestamp", -1, "timestamp of the object"};
51 
52  void init(o2::framework::InitContext&)
53  {
54  ccdb->setURL(url.value);
55  ccdb->setTimestamp(timestamp.value);
56  ccdb->setCaching(true);
57  ccdb->setLocalObjectValidityChecking();
58  // Not later than now objects
59  ccdb->setCreatedNotAfter(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
60  //
61  const std::string fname = paramfile.value;
62  if (!fname.empty()) { // Loading the parametrization from file
63  resp.LoadParamFromFile(fname.data(), signalname.value, DetectorResponse::kSignal);
64  resp.LoadParamFromFile(fname.data(), sigmaname.value, DetectorResponse::kSigma);
65  } else { // Loading it from CCDB
66  const std::string path = "Analysis/PID/TPC";
67  resp.LoadParam(DetectorResponse::kSignal, ccdb->getForTimeStamp<Parametrization>(path + "/" + signalname.value, timestamp.value));
68  resp.LoadParam(DetectorResponse::kSigma, ccdb->getForTimeStamp<Parametrization>(path + "/" + sigmaname.value, timestamp.value));
69  }
70  }
71 
72  void process(Coll const& collisions, Trks const& tracks)
73  {
74  constexpr tpc::ELoss<Coll::iterator, Trks::iterator, pid_type> resp_PID = tpc::ELoss<Coll::iterator, Trks::iterator, pid_type>();
75 
76  tpcpid.reserve(tracks.size());
77  for (auto const& trk : tracks) {
78  tpcpid(resp_PID.GetExpectedSigma(resp, trk.collision(), trk),
79  resp_PID.GetSeparation(resp, trk.collision(), trk));
80  }
81  }
82 };

84 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
85 {
86  WorkflowSpec workflow;
87  if (cfgc.options().get<int>("pid-el")) {
88  workflow.push_back(adaptAnalysisTask<pidTPCTaskPerParticle<PID::Electron, o2::aod::pidRespTPCEl>>("pidTPCEl-task"));
89  }
90  if (cfgc.options().get<int>("pid-mu")) {
91  workflow.push_back(adaptAnalysisTask<pidTPCTaskPerParticle<PID::Muon, o2::aod::pidRespTPCMu>>("pidTPCMu-task"));
92  }
93  if (cfgc.options().get<int>("pid-pikapr")) {
94  workflow.push_back(adaptAnalysisTask<pidTPCTaskPerParticle<PID::Pion, o2::aod::pidRespTPCPi>>("pidTPCPi-task"));
95  workflow.push_back(adaptAnalysisTask<pidTPCTaskPerParticle<PID::Kaon, o2::aod::pidRespTPCKa>>("pidTPCKa-task"));
96  workflow.push_back(adaptAnalysisTask<pidTPCTaskPerParticle<PID::Proton, o2::aod::pidRespTPCPr>>("pidTPCPr-task"));
97  }
98  if (cfgc.options().get<int>("pid-nuclei")) {
99  workflow.push_back(adaptAnalysisTask<pidTPCTaskPerParticle<PID::Deuteron, o2::aod::pidRespTPCDe>>("pidTPCDe-task"));
100  workflow.push_back(adaptAnalysisTask<pidTPCTaskPerParticle<PID::Triton, o2::aod::pidRespTPCTr>>("pidTPCTr-task"));
101  workflow.push_back(adaptAnalysisTask<pidTPCTaskPerParticle<PID::Helium3, o2::aod::pidRespTPCHe>>("pidTPCHe-task"));
102  workflow.push_back(adaptAnalysisTask<pidTPCTaskPerParticle<PID::Alpha, o2::aod::pidRespTPCAl>>("pidTPCAl-task"));
103  }
104  return workflow;
105 }
