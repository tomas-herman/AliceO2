176  aod::McCollisions const& mccollisions,
177  soa::Join<aod::HfCandProng3, aod::HfCandProng3MCRec, aod::HFSelLcCandidate> const& candidates,
178  soa::Join<aod::McParticles, aod::HfCandProng3MCGen> const& particles,
179  aod::BigTracksPID const& tracks)

186  collision.posX(),

188  collision.posZ(),
189  0,
190  1);
191  }
192  rowCandidateFull.reserve(candidates.size());
193  for (auto& candidate : candidates) {
194 #define FILL_TREE(CandFlag, FunctionSelection, FunctionInvMass, FunctionCt, FunctionY, FunctionE) \
195  if (candidate.FunctionSelection() >= 1) { \
196  rowCandidateFull( \
197  candidate.index0_as<aod::BigTracksPID>().collision().bcId(), \
198  candidate.index0_as<aod::BigTracksPID>().collision().numContrib(), \
199  candidate.posX(), \
200  candidate.posY(), \
201  candidate.posZ(), \
202  candidate.xSecondaryVertex(), \

208  candidate.rSecondaryVertex(), \
209  candidate.decayLength(), \
210  candidate.decayLengthXY(), \
211  candidate.decayLengthNormalised(), \
212  candidate.decayLengthXYNormalised(), \
213  candidate.impactParameterNormalised0(), \
214  TMath::Abs(candidate.ptProng0()), \
215  TMath::Sqrt(RecoDecay::P(candidate.pxProng0(), candidate.pyProng0(), candidate.pzProng0())), \
216  candidate.impactParameterNormalised1(), \
217  TMath::Abs(candidate.ptProng1()), \
218  TMath::Sqrt(RecoDecay::P(candidate.pxProng1(), candidate.pyProng1(), candidate.pzProng1())), \
219  candidate.impactParameterNormalised2(), \
220  TMath::Abs(candidate.ptProng2()), \
221  TMath::Sqrt(RecoDecay::P(candidate.pxProng2(), candidate.pyProng2(), candidate.pzProng2())), \
222  candidate.pxProng0(), \
223  candidate.pyProng0(), \
224  candidate.pzProng0(), \
225  candidate.pxProng1(), \
226  candidate.pyProng1(), \
227  candidate.pzProng1(), \
228  candidate.pxProng2(), \
229  candidate.pyProng2(), \
230  candidate.pzProng2(), \
231  candidate.impactParameter0(), \
232  candidate.impactParameter1(), \
233  candidate.impactParameter2(), \
234  candidate.errorImpactParameter0(), \
235  candidate.errorImpactParameter1(), \
236  candidate.errorImpactParameter2(), \
237  candidate.index0_as<aod::BigTracksPID>().tpcNSigmaPi(), \
238  candidate.index0_as<aod::BigTracksPID>().tpcNSigmaKa(), \
239  candidate.index0_as<aod::BigTracksPID>().tofNSigmaPi(), \
240  candidate.index0_as<aod::BigTracksPID>().tofNSigmaKa(), \
241  candidate.index1_as<aod::BigTracksPID>().tpcNSigmaPi(), \
242  candidate.index1_as<aod::BigTracksPID>().tpcNSigmaKa(), \
243  candidate.index1_as<aod::BigTracksPID>().tofNSigmaPi(), \
244  candidate.index1_as<aod::BigTracksPID>().tofNSigmaKa(), \
245  candidate.index2_as<aod::BigTracksPID>().tpcNSigmaPi(), \
246  candidate.index2_as<aod::BigTracksPID>().tpcNSigmaKa(), \
247  candidate.index2_as<aod::BigTracksPID>().tofNSigmaPi(), \
248  candidate.index2_as<aod::BigTracksPID>().tofNSigmaKa(), \
249  1 << CandFlag, \
250  FunctionInvMass(candidate), \
251  candidate.pt(), \
252  candidate.p(), \
253  candidate.cpa(), \
254  candidate.cpaXY(), \
255  FunctionCt(candidate), \
256  candidate.eta(), \
257  candidate.phi(), \
258  FunctionY(candidate), \
259  FunctionE(candidate), \
260  candidate.flagMCMatchRec()); \
261  }
262 
263  FILL_TREE(0, isSelLcpKpi, InvMassLcpKpi, CtLc, YLc, ELc);
264  FILL_TREE(1, isSelLcpiKp, InvMassLcpiKp, CtLc, YLc, ELc);
265  }
266  int npart = 0;
267  rowCandidateFullParticles.reserve(particles.size());
268  for (auto& particle : particles) {
269  if (particle.flagMCMatchGen()) {
270  rowCandidateFullParticles(
271  particle.mcCollision().bcId(),
272  particle.pt(),
273  particle.eta(),
274  particle.phi(),
275  RecoDecay::Y(array{particle.px(), particle.py(), particle.pz()}, RecoDecay::getMassPDG(particle.pdgCode())),
276  particle.flagMCMatchGen());
277  }
278  }
279  }
