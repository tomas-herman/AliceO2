44 class TPCCalibPedestalDevice : public o2::framework::Task
45 {
46  public:
47  TPCCalibPedestalDevice(bool skipCalib) : mSkipCalib(skipCalib) {}
48 
49  void init(o2::framework::InitContext& ic) final
50  {
51  // set up ADC value filling
52  mCalibPedestal.init(); // initialize configuration via configKeyValues
53  mRawReader.createReader("");
54  mRawReader.setADCDataCallback([this](const PadROCPos& padROCPos, const CRU& cru, const gsl::span<const uint32_t> data) -> int {
55  const int timeBins = mCalibPedestal.update(padROCPos, cru, data);
56  mCalibPedestal.setNumberOfProcessedTimeBins(std::max(mCalibPedestal.getNumberOfProcessedTimeBins(), size_t(timeBins)));
57  return timeBins;
58  });
59 
60  mMaxEvents = static_cast<uint32_t>(ic.options().get<int>("max-events"));
61  mUseOldSubspec = ic.options().get<bool>("use-old-subspec");
62  mForceQuit = ic.options().get<bool>("force-quit");
63  mDirectFileDump = ic.options().get<bool>("direct-file-dump");
64  if (mUseOldSubspec) {
65  LOGP(info, "Using old subspecification (CruId << 16) | ((LinkId + 1) << (CruEndPoint == 1 ? 8 : 0))");
66  }
67  }
68 
69  void run(o2::framework::ProcessingContext& pc) final
70  {
71  // in case the maximum number of events was reached don't do further processing
72  if (mReadyToQuit) {
73  return;
74  }
75 
76  auto& reader = mRawReader.getReaders()[0];
77  calib_processing_helper::processRawData(pc.inputs(), reader, mUseOldSubspec);
78 
79  mCalibPedestal.incrementNEvents();
80  LOGP(info, "Number of processed events: {} ({})", mCalibPedestal.getNumberOfProcessedEvents(), mMaxEvents);
81 
82  if ((mCalibPedestal.getNumberOfProcessedEvents() >= mMaxEvents) && !mCalibDumped) {
83  LOGP(info, "Maximm number of events reached ({}), no more processing will be done", mMaxEvents);
84  mReadyToQuit = true;
85  dumpCalibData();
86  if (mForceQuit) {
87  pc.services().get<ControlService>().endOfStream();
88  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
89  } else {
90  //pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
91  }
92  }
93  }
94 
95  void endOfStream(o2::framework::EndOfStreamContext& ec) final
96  {
97  LOGP(info, "endOfStream");
98  dumpCalibData();
99  if (!mSkipCalib) {
100  sendOutput(ec.outputs());
101  }
102  ec.services().get<ControlService>().readyToQuit(QuitRequest::Me);
103  }
104 
105  private:
106  CalibPedestal mCalibPedestal;
107  rawreader::RawReaderCRUManager mRawReader;
108  uint32_t mMaxEvents{100};
109  bool mReadyToQuit{false};
110  bool mCalibDumped{false};
111  bool mUseOldSubspec{false};
112  bool mForceQuit{false};
113  bool mDirectFileDump{false};
114  bool mSkipCalib{false};

116  //____________________________________________________________________________
117  void sendOutput(DataAllocator& output)
118  {
119  CDBStorage::MetaData_t md;
120 
121  // perhaps should be changed to time of the run
122  const auto now = std::chrono::system_clock::now();
123  long timeStart = std::chrono::duration_cast<std::chrono::seconds>(now.time_since_epoch()).count();
124  long timeEnd = 99999999999999;
125 
126  std::array<const CalDet<float>*, 2> data = {&mCalibPedestal.getPedestal(), &mCalibPedestal.getNoise()};
127  std::array<CDBType, 2> dataType = {CDBType::CalPedestal, CDBType::CalNoise};
128 
129  for (size_t i = 0; i < data.size(); ++i) {
130  auto cal = data[i];
131  o2::ccdb::CcdbObjectInfo w;
132  auto image = o2::ccdb::CcdbApi::createObjectImage(cal, &w);
133 
134  w.setPath(CDBTypeMap.at(dataType[i]));
135  w.setStartValidityTimestamp(timeStart);
136  w.setEndValidityTimestamp(timeEnd);
137 
138  LOG(INFO) << "Sending object " << w.getPath() << "/" << w.getFileName() << " of size " << image->size()
139  << " bytes, valid for " << w.getStartValidityTimestamp() << " : " << w.getEndValidityTimestamp();
140 
141  header::DataHeader::SubSpecificationType subSpec{(header::DataHeader::SubSpecificationType)i};
142  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload, subSpec}, *image.get());
143  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo, subSpec}, w);
144  }
145  }

147  //____________________________________________________________________________
148  void dumpCalibData()
149  {
150  if (mDirectFileDump && !mCalibDumped) {
151  LOGP(info, "Dumping output");
152  mCalibPedestal.analyse();
153  mCalibPedestal.dumpToFile("pedestals.root");
154  mCalibDumped = true;
155  }
156  }
157 };

164 DataProcessorSpec getTPCCalibPedestalSpec(const std::string inputSpec, bool skipCalib)
165 {
166  using device = o2::calibration::TPCCalibPedestalDevice;
167 
168  std::vector<OutputSpec> outputs;
169  if (!skipCalib) {
170  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload});
171  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo});
172  }
173 
174  return DataProcessorSpec{
175  "calib-tpc-pedestal",
176  select(inputSpec.data()),
177  outputs,
178  AlgorithmSpec{adaptFromTask<device>(skipCalib)},
179  Options{
180  {"max-events", VariantType::Int, 100, {"maximum number of events to process"}},
181  {"use-old-subspec", VariantType::Bool, false, {"use old subsecifiation definition"}},
182  {"force-quit", VariantType::Bool, false, {"force quit after max-events have been reached"}},
183  {"direct-file-dump", VariantType::Bool, false, {"directly dump calibration to file"}},
184  } // end Options
185  }; // end DataProcessorSpec
