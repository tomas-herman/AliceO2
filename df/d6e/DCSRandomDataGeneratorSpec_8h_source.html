56 std::vector<int> generateIntegers(size_t size, int min, int max)
57 {
58  std::uniform_int_distribution<int> distribution(min, max);
59  std::mt19937 generator(std::random_device{}());
60  std::vector<int> data;
61  while (data.size() != size) {
62  data.emplace_back(distribution(generator));
63  std::sort(begin(data), end(data));
64  auto last = std::unique(begin(data), end(data)); // make sure we do not duplicate
65  data.erase(last, end(data));
66  }
67  std::shuffle(begin(data), end(data), generator);
68  for (auto i = 0; i < data.size(); ++i) {
69  LOG(INFO) << "Generating randomly DP at index " << data[i];
70  }
71  return data;
72 }

81 std::vector<o2::dcs::DataPointCompositeObject> generate(const std::vector<HintType> hints,
82  float fraction = 1.0,
83  uint64_t tfid = 0)
84 {
85  std::vector<o2::dcs::DataPointCompositeObject> dataPoints;
86 
87  TDatime d;
88  auto dsec = d.Convert();
89  dsec += tfid;
90  d.Set(dsec);
91 
92  std::string refDate = d.AsString();
93 
94  auto GenerateVisitor = [refDate](const auto& t) {
95  return o2::dcs::generateRandomDataPoints({t.aliasPattern}, t.minValue, t.maxValue, refDate);
96  };
97 
98  for (const auto& hint : hints) {
99  auto dpcoms = std::visit(GenerateVisitor, hint);
100  for (auto dp : dpcoms) {
101  dataPoints.push_back(dp);
102  }
103  }
104  if (fraction < 1.0) {
105  auto indices = generateIntegers(fraction * dataPoints.size(), 0, dataPoints.size() - 1);
106  std::vector<o2::dcs::DataPointCompositeObject> tmp;
107  tmp.swap(dataPoints);
108  dataPoints.clear();
109  for (auto i : indices) {
110  dataPoints.push_back(tmp[i]);
111  }
112  }
113  return dataPoints;
114 }

162  void run(o2::framework::ProcessingContext& pc) final
163  {
164  auto input = pc.inputs().begin();
165  uint64_t tfid = o2::header::get<o2::framework::DataProcessingHeader*>((*input).header)->startTime;
166  if (tfid >= mMaxTF) {
167  LOG(INFO) << "Data generator reached TF " << tfid << ", stopping";
168  pc.services().get<o2::framework::ControlService>().endOfStream();
169  pc.services().get<o2::framework::ControlService>().readyToQuit(o2::framework::QuitRequest::Me);
170  }
171 
172  bool generateFBI = (mTFs % mMaxCyclesNoFullMap == 0);
173  // fraction is one if we generate FBI (Full Buffer Image)
174  float fraction = (generateFBI ? 1.0 : mDeltaFraction);
175 
176  TDatime d;
177  auto dpcoms = generate(mDataPointHints, fraction, tfid);
178  auto tofdpcoms = generate(mTOFDataPointHints, fraction, tfid);
179 
180  LOG(INFO) << "***************** TF " << tfid << " has generated " << tofdpcoms.size() << " DPs for TOF";
181  pc.outputs().snapshot(Output{"DCS", "DATAPOINTS", 0, Lifetime::Timeframe}, dpcoms);
182  pc.outputs().snapshot(Output{"DCS", "TOFDATAPOINTS", 0, Lifetime::Timeframe}, tofdpcoms);
183  mTFs++;
184  }
185

188  uint64_t mTFs = 0;
189  uint64_t mMaxCyclesNoFullMap;
190  float mDeltaFraction;
191  std::vector<HintType> mDataPointHints;
192  std::vector<HintType> mTOFDataPointHints;
