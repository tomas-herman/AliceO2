104 template <class T>
105 GPUdi() void GPUTPCCompression::truncateSignificantBits(T& v, unsigned int nBits, unsigned int max)
106 {
107  if (nBits == 0) {
108  return;
109  }
110 
111  unsigned int val = v;
112  unsigned int ldz = sizeof(unsigned int) * 8 - CAMath::Clz(val);
113  if (val && ldz > nBits) {
114  if (val & (1 << (ldz - nBits - 1))) {
115  val += (1 << (ldz - nBits - 1));
116  ldz = sizeof(unsigned int) * 8 - CAMath::Clz(val);
117  }
118  val &= ((1 << ldz) - 1) ^ ((1 << (ldz - nBits)) - 1);
119  if (val >= max) {
120  val = max - 1;
121  }
122  // GPUInfo("CHANGING X %x --> %x", (unsigned int) v, val);
123  v = val;
124  }
125 }
