38 struct HFCandidateCreator2Prong {
39  Produces<aod::HfCandProng2Base> rowCandidateBase;
40 
41  Configurable<double> magneticField{"d_bz", 5., "magnetic field"};
42  Configurable<bool> b_propdca{"b_propdca", true, "create tracks version propagated to PCA"};
43  Configurable<double> d_maxr{"d_maxr", 200., "reject PCA's above this radius"};
44  Configurable<double> d_maxdzini{"d_maxdzini", 4., "reject (if>0) PCA candidate if tracks DZ exceeds threshold"};
45  Configurable<double> d_minparamchange{"d_minparamchange", 1.e-3, "stop iterations if largest change of any X is smaller than this"};
46  Configurable<double> d_minrelchi2change{"d_minrelchi2change", 0.9, "stop iterations is chi2/chi2old > this"};
47  Configurable<bool> b_dovalplots{"b_dovalplots", true, "do validation plots"};
48 
49  OutputObj<TH1F> hmass2{TH1F("hmass2", "2-prong candidates;inv. mass (#pi K) (GeV/#it{c}^{2});entries", 500, 0., 5.)};
50  OutputObj<TH1F> hCovPVXX{TH1F("hCovPVXX", "2-prong candidates;XX element of cov. matrix of prim. vtx. position (cm^{2});entries", 100, 0., 1.e-4)};
51  OutputObj<TH1F> hCovSVXX{TH1F("hCovSVXX", "2-prong candidates;XX element of cov. matrix of sec. vtx. position (cm^{2});entries", 100, 0., 0.2)};
52 
53  double massPi = RecoDecay::getMassPDG(kPiPlus);
54  double massK = RecoDecay::getMassPDG(kKPlus);
55  double massPiK{0.};
56  double massKPi{0.};
57 
58  void process(aod::Collisions const& collisions,
59  aod::HfTrackIndexProng2 const& rowsTrackIndexProng2,
60  aod::BigTracks const& tracks)
61  {
62  // 2-prong vertex fitter
63  o2::vertexing::DCAFitterN<2> df;
64  df.setBz(magneticField);
65  df.setPropagateToPCA(b_propdca);
66  df.setMaxR(d_maxr);
67  df.setMaxDZIni(d_maxdzini);
68  df.setMinParamChange(d_minparamchange);
69  df.setMinRelChi2Change(d_minrelchi2change);
70  df.setUseAbsDCA(true);
71 
72  // loop over pairs of track indeces
73  for (const auto& rowTrackIndexProng2 : rowsTrackIndexProng2) {
74  auto trackParVarPos1 = getTrackParCov(rowTrackIndexProng2.index0());
75  auto trackParVarNeg1 = getTrackParCov(rowTrackIndexProng2.index1());
76  auto collision = rowTrackIndexProng2.index0().collision();
77 
78  // reconstruct the 2-prong secondary vertex
79  if (df.process(trackParVarPos1, trackParVarNeg1) == 0) {
80  continue;
81  }
82  const auto& secondaryVertex = df.getPCACandidate();
83  auto chi2PCA = df.getChi2AtPCACandidate();
84  auto covMatrixPCA = df.calcPCACovMatrix().Array();
85  hCovSVXX->Fill(covMatrixPCA[0]); // FIXME: Calculation of errorDecayLength(XY) gives wrong values without this line.
86  auto trackParVar0 = df.getTrack(0);
87  auto trackParVar1 = df.getTrack(1);
88 
89  // get track momenta
90  array<float, 3> pvec0;
91  array<float, 3> pvec1;
92  trackParVar0.getPxPyPzGlo(pvec0);
93  trackParVar1.getPxPyPzGlo(pvec1);
94 
95  // get track impact parameters
96  // This modifies track momenta!
97  auto primaryVertex = getPrimaryVertex(collision);
98  auto covMatrixPV = primaryVertex.getCov();
99  hCovPVXX->Fill(covMatrixPV[0]);
100  o2::dataformats::DCA impactParameter0;
101  o2::dataformats::DCA impactParameter1;
102  trackParVar0.propagateToDCA(primaryVertex, magneticField, &impactParameter0);
103  trackParVar1.propagateToDCA(primaryVertex, magneticField, &impactParameter1);
104 
105  // get uncertainty of the decay length
106  double phi, theta;
107  getPointDirection(array{collision.posX(), collision.posY(), collision.posZ()}, secondaryVertex, phi, theta);
108  auto errorDecayLength = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, theta) + getRotatedCovMatrixXX(covMatrixPCA, phi, theta));
109  auto errorDecayLengthXY = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, 0.) + getRotatedCovMatrixXX(covMatrixPCA, phi, 0.));
110 
111  // fill candidate table rows
112  rowCandidateBase(collision.posX(), collision.posY(), collision.posZ(),
113  secondaryVertex[0], secondaryVertex[1], secondaryVertex[2],
114  errorDecayLength, errorDecayLengthXY,
115  chi2PCA,
116  pvec0[0], pvec0[1], pvec0[2],
117  pvec1[0], pvec1[1], pvec1[2],
118  impactParameter0.getY(), impactParameter1.getY(),
119  std::sqrt(impactParameter0.getSigmaY2()), std::sqrt(impactParameter1.getSigmaY2()),
120  rowTrackIndexProng2.index0Id(), rowTrackIndexProng2.index1Id(),
121  rowTrackIndexProng2.hfflag());
122 
123  // fill histograms
124  if (b_dovalplots) {
125  // calculate invariant masses
126  auto arrayMomenta = array{pvec0, pvec1};
127  massPiK = RecoDecay::M(arrayMomenta, array{massPi, massK});
128  massKPi = RecoDecay::M(arrayMomenta, array{massK, massPi});
129  hmass2->Fill(massPiK);
130  hmass2->Fill(massKPi);
131  }
132  }
133  }
134 };

143 struct HFCandidateCreator2ProngMC {
144  Produces<aod::HfCandProng2MCRec> rowMCMatchRec;
145  Produces<aod::HfCandProng2MCGen> rowMCMatchGen;
146 
147  void process(aod::HfCandProng2 const& candidates,
148  aod::BigTracksMC const& tracks,
149  aod::McParticles const& particlesMC)
150  {
151  int8_t sign = 0;
152  int8_t result = N2ProngDecays;
153 
154  // Match reconstructed candidates.
155  for (auto& candidate : candidates) {
156  //Printf("New rec. candidate");
157  result = N2ProngDecays;
158  auto arrayDaughters = array{candidate.index0_as<aod::BigTracksMC>(), candidate.index1_as<aod::BigTracksMC>()};
159 
160  // D0(bar) → π± K∓
161  //Printf("Checking D0(bar) → π± K∓");
162  if (RecoDecay::getMatchedMCRec(particlesMC, std::move(arrayDaughters), 421, array{+kPiPlus, -kKPlus}, true, &sign) > -1) {
163  result = sign * D0ToPiK;
164  }
165  // Jpsi → e+e-
166  if (result == N2ProngDecays) {
167  //Printf("Checking Jpsi → e+e-");
168  if (RecoDecay::getMatchedMCRec(particlesMC, std::move(arrayDaughters), 443, array{+kElectron, -kElectron}, true, &sign) > -1) {
169  result = sign * JpsiToEE;
170  }
171  }
172 
173  rowMCMatchRec(result);
174  }
175 
176  // Match generated particles.
177  for (auto& particle : particlesMC) {
178  //Printf("New gen. candidate");
179  result = N2ProngDecays;
180 
181  // D0(bar) → π± K∓
182  //Printf("Checking D0(bar) → π± K∓");
183  if (RecoDecay::isMatchedMCGen(particlesMC, particle, 421, array{+kPiPlus, -kKPlus}, true, &sign)) {
184  result = sign * D0ToPiK;
185  }
186  // Jpsi → e+e-
187  if (result == N2ProngDecays) {
188  //Printf("Checking Jpsi → e+e-");
189  if (RecoDecay::isMatchedMCGen(particlesMC, particle, 443, array{+kElectron, -kElectron}, true, &sign)) {
190  result = sign * JpsiToEE;
191  }
192  }
193 
194  rowMCMatchGen(result);
195  }
196  }
197 };
